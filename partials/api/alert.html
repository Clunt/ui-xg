Alert

Description

提示组件,可展示不同样式的提示信息。

Usage

<fugu-alert text="string"
    [ close="boolean" ]
    [ close-func="string" ]
    [ close-text="string" ]
    [ has-icon="boolean" ]
    [ dismiss-on-timeout="number" ]
    [ template-url="string" ]
    [ type="string" ]>
</fugu-button>

Restrict

  • 'E'

Arguments

ParamTypeDefaultDetail
close(optional) false指定提示是否可关闭,默认显示false。
closeFunc(optional) null指定提示关闭的回调函数,取父作用域上的函数,默认为自带回调函数。
closeText(optional) null设置关闭按钮的替代文案,不适用则不显示文案。
hasIcon(optional) false设置是否使用图标。
dismissOnTimeout(optional) null设置提示自动关闭的时间，单位是毫秒。要使用这个属性必须close这个属性存在。
templateUrl(optional) null指定提示使用自定义模板的路径。默认不适用自定义模板。
type(optional) warning设置提示的类别，有danger、success、warning和info四种可选，默认为warning。

Example

  • index.html
  • script.js
<div ng-controller="alertDemoCtrl">
    <fugu-alert ng-repeat="alert in alerts" type="{{alert.type}}" close="true" close-func="closeAlert($index)" has-icon="true">{{alert.msg}}</fugu-alert>
    <fugu-alert type="warning" close="true" dismiss-on-timeout="3000" has-icon="true">3s定时关闭</fugu-alert>
    <fugu-alert type="warning" close="true" close-text="点击关闭" has-icon="true">替换关闭按钮文案</fugu-alert>
    <fugu-alert type="warning" close="true" close-text="点击关闭" has-icon="{{true}}">替换关闭按钮文案</fugu-alert>
    <fugu-alert type="warning" close="false">取消图标</fugu-alert>
    <fugu-alert type="warning" close="{{boolean}}">布尔取值为{{boolean}}</fugu-alert>
    <fugu-alert template-url="alert.html">A happy alert!</fugu-alert>

    <button ng-click="addAlert()">点击添加</button>
</div>
angular.module('fuguDemo').controller('alertDemoCtrl',['$scope', function ($scope) {
    $scope.isIcon = false;
    $scope.boolean = true;
    $scope.alerts = [
        { type: 'danger', msg: 'Oh snap! Change a few things up and try submitting again.' },
        { type: 'success', msg: 'Well done! You successfully read this important alert message.' },
        { type: 'warning', msg: 'FBI Warning! Manong would ignore anything about warning.' },
        { type: 'info', msg: 'I know that you wouldn\'t see this line.' }
    ];
    $scope.closeAlert = function(index) {
        $scope.alerts.splice(index, 1);
    };
    $scope.addAlert = function(){
        $scope.alerts.push({msg:"this is a new alert."});
    }
}]);
