pager

Description

search box directive.

Usage

<fugu-search-box [ng-model="string"]
    placeholder="test">
    </fugu-search-box>

Restrict

  • 'AE'

Arguments

ParamTypeDefaultDetails
showBtn(optional) true是否展示搜索按钮
btnText(optional) "搜索"展示搜索按钮的话，指定按钮上的文本
search(optional) none在搜索框中回车键或者点击搜索按钮之后触发

Example

  • index.html
  • script.js
<div ng-controller="pagerDemoCtrl">
    <fugu-pager page-no="pages.pageNo" items-per-page="pages.pageSize" next-text="Next" total-items="pages.totalCount"></fugu-pager>
    <br/>
    <fugu-pager page-no="pages.pageNo" total-items="pages.totalCount" first-text="First" last-text="Last" page-changed="changePage"></fugu-pager>
    <br/>
    <pre>currentPageNo is : {{pages.pageNo}}</pre>
    <label for="setPage">设置页码:</label><input id="setPage" type="number" min="1" ng-change="setPageNo(pages.pageNo)" ng-model="pages.pageNo"/>
</div>
angular.module('fuguDemo').controller('pagerDemoCtrl',['$scope', function ($scope) {
    $scope.pages = {
        pageNo:1,
        pageSize:30,
        totalCount:110
    };
    $scope.setPageNo = function (val) {
        $scope.pages.pageNo = val;
    };
    $scope.changePage = function (pageNo) {
        console.log('修改页码为:'+pageNo);
    }
}]);
