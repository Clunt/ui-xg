pager

Description

分页器组件

Usage

<uix-pager total-items="number"
    ng-model="number"
    [ items-per-page="number" ]
    [ max-size="number" ]
    [ first-text="{{string}}" ]
    [ last-text="{{string}}" ]
    [ previous-text="{{string}}" ]
    [ next-text="{{string}}" ]
    [ boundary-links="boolean" ]
    [ direction-links="boolean" ]
    [ show-total="boolean" ]>
    </uix-pager>

Restrict

  • 'E'

Arguments

ParamTypeDefaultDetail
totalItems 条目总数
ngModel 当前页码,可以用ng-change监控变化
itemsPerPage(optional) 20每一页的数量
maxSize(optional) 5分页器展示页码的最大数量
firstText(optional) "首页"首页按钮文本
lastText(optional) "尾页"尾页按钮文本
previousText(optional) "上一页"上一页按钮文本
nextText(optional) "下一页"下一页按钮文本
boundaryLinks(optional) true是否显示边界的按钮,默认是"首页"和"尾页"按钮
directionLinks(optional) true是否显示方向的按钮,默认是"上一页"和"下一页"按钮
showTotal(optional) true是否显示条目和页码总数

<div ng-controller="pagerDemoCtrl">
    <h4>default</h4>
    <uix-pager total-items="totalItems" ng-model="currentPage" ng-change="pageChanged()"></uix-pager>
    <h4>text</h4>
    <uix-pager total-items="totalItems" ng-model="currentPage" class="pagination-sm"
               previous-text="Previous" next-text="Next" first-text="First" last-text="Last"></uix-pager>
    <h4>directionLinks and boundaryLinks</h4>
    <uix-pager direction-links="false" total-items="totalItems" ng-model="currentPage"></uix-pager>
    <uix-pager boundary-links="false" total-items="totalItems" ng-model="currentPage"></uix-pager>
    <pre>The selected page no: {{currentPage}}</pre>
    <button class="btn btn-info" ng-click="setPage(3)">Set current page to: 3</button>
    <hr>
    <h4>Limit the maximum visible buttons</h4>
    <uix-pager total-items="bigTotalItems" ng-model="bigCurrentPage" max-size="maxSize" class="pagination-sm"></uix-pager>
    <uix-pager total-items="bigTotalItems" ng-model="bigCurrentPage" max-size="maxSize" class="pagination-sm" rotate="false"></uix-pager>
    <hr>
    <h4>do not show total info</h4>
    <uix-pager total-items="totalItems" ng-model="bigCurrentPage" show-total="false"></uix-pager>
</div>
angular.module('uixDemo').controller('pagerDemoCtrl', ['$scope', '$log', function ($scope, $log) {
    $scope.totalItems = 64;
    $scope.currentPage = 4;

    $scope.setPage = function (pageNo) {
        $scope.currentPage = pageNo;
    };

    $scope.pageChanged = function () {
        $log.log('Page changed to: ' + $scope.currentPage);
    };

    $scope.maxSize = 5;
    $scope.bigTotalItems = 175;
    $scope.bigCurrentPage = 1;
    $scope.$on('uixPager:pageChanged', function (evt, arg) {
        $log.log('pageChanged event:' + arg);
    });
}]);
