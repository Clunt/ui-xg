tree

Description

树组件,可以用于以层级的形式展示数据,也可以用于选择数据。

Usage

<fugu-tree ng-model="array"
    [ show-icon="boolean" ]
    [ checkable="boolean" ]
    [ expand-all="boolean" ]
    [ on-click="function" ]
    [ on-check-change="function" ] >
</fugu-tree>

Restrict

  • 'AE'

Arguments

ParamTypeDefaultDetail
ngModel 绑定tree展示和操作数据,具有特定格式[{label: '省份', children: []}],其中如果是叶子节点,children为空,可以省略,如果不为空其内容也是上述格式.
showIcon(optional) true是否显示文件夹|文件图标,默认显示
checkable(optional) true是否显示勾选框,默认显示
expandAll(optional) true是否展开树,默认展开,当点击树节点时
onClick(optional) 点击树节点名称时触发的事件,同时可以派发出一个参数data(表示点击节点)
onCheckChange(optional) 勾选|取消勾选时触发的事件,同时可以派发出一个参数data(表示选中节点集)

Example

  • index.html
  • script.js
  • style.css
<div ng-controller="treeDemoCtrl">
    <ul class="treeDemo">
        <li>
            <p>ng-model绑定数据:数据有格式要求</p>
            <fugu-tree ng-model="nodes"></fugu-tree>
        </li>
        <li>
            <p>show-icon设置是否显示图标</p>
            <fugu-tree show-icon="showIcon" ng-model="nodes"></fugu-tree>
        </li>
        <li>
            <p>checkable设置是否可勾选</p>
            <fugu-tree checkable="checkable" ng-model="nodes"></fugu-tree>
        </li>
        <li>
            <p>expand-all设置是否展开</p>
            <fugu-tree expand-all="expandAll" ng-model="nodes"></fugu-tree>
        </li>
        <li>
            <p>on-click和on-check-change事件</p>
            <fugu-tree on-click="clickFn" on-check-change="checkChangeFn" ng-model="nodes"></fugu-tree>
            <div ng-if="selectData">selectData:<pre>{{selectData | json}}</pre></div>
            <div ng-if="clickData">clickData:<pre>{{clickData | json}}</pre></div>
        </li>
    </ul>
</div>
angular.module('fuguDemo').controller('treeDemoCtrl',['$scope', function ($scope) {
    $scope.nodes = [
        {label: '省份', children: [
            {label: '陕西', children:[
                {label: '汉中'},
                {label: '勉县'},
                {label: '咸阳'}
            ]},
            {label: '北京', children:[
                {label: '朝阳'},
                {label: '昌平'},
                {label: '顺义'}
            ]},
            {label: '浙江', children:[
                {label: '宁波'},
                {label: '杭州'},
                {label: '嘉兴'}
            ]}
        ]}
    ];
    $scope.expandAll = false;
    $scope.checkable = false;
    $scope.showIcon = false;
    $scope.clickFn = function(data){
        $scope.clickData = data;
    };
    $scope.checkChangeFn = function(data){
        $scope.selectData = data;
    };
}]);
