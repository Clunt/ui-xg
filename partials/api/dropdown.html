dropdown

Description

下拉菜单组件，支持多列。可以通过Provider设置多列数目，当列表项的数目大于多列数目时以多列显示

Usage

<fugu-dropdown
    [ is-open="boolean" ]
    [ cols-num="number" ]>
        <button fugu-dropdown-toggle type="button" class="btn">Text</button>
        <fugu-dropdown-choices>{{string}}</fugu-dropdown-choices>
    ...
</fugu-dropdown>

Restrict

  • 'E'

Provider

  • fuguDropdownProvider：全局配置dropdown相关的设置
  • methods
    • setColsNum(num)：设置多列数目

Arguments

ParamTypeDefaultDetail
isOpen false是否开启列表
colsNum 3设置列数,属性的配置会覆盖全局provider的配置,全局的配置会覆盖默认的配置

Example

  • index.html
  • script.js
<div ng-controller="dropdownDemoCtrl">
    <fugu-dropdown cols-num="1">
        <button fugu-dropdown-toggle type="button" class="btn btn-sm btn-primary">
            {{currentUser.name}}&nbsp;<span class="caret"></span>
        </button>
        <fugu-dropdown-choices ng-repeat="item in list" title="{{item.name}}" ng-click="selectItem(item)">{{item.name}}</fugu-dropdown-choices>
    </fugu-dropdown>
    <fugu-dropdown is-open="status.isOpen">
        <button fugu-dropdown-toggle type="button" ng-disabled="status.isDisabled" class="btn btn-sm btn-primary">
            {{currentUser.name}}&nbsp;<span class="caret"></span>
        </button>
        <fugu-dropdown-choices title="第一个">第一个</fugu-dropdown-choices>
        <fugu-dropdown-choices title="第二项">第二项</fugu-dropdown-choices>
        <fugu-dropdown-choices title="第三项">第三项</fugu-dropdown-choices>
        <fugu-dropdown-choices title="第四项">第四项</fugu-dropdown-choices>
        <fugu-dropdown-choices title="第五项">第五项</fugu-dropdown-choices>
    </fugu-dropdown>
    isOpen:{{status.isOpen}}<input type="checkbox" ng-model="status.isOpen"/>
    isDisabled:{{status.isDisabled}}<input type="checkbox" ng-model="status.isDisabled"/>
    <button type="button" class="btn btn-default" ng-click="doClick($event)">点击</button>
</div>
angular.module('fuguDemo').config(['fuguDropdownProvider', function (fuguDropdownProvider) {
    fuguDropdownProvider.setColsNum(3);
}]);
angular.module('fuguDemo').controller('dropdownDemoCtrl',['$scope', function ($scope) {
    $scope.list = [
        {id:1,name:'Jerry'},
        {id:2,name:'Test'},
        {id:3,name:'multidropdownmultidropdown'},
        {id:4,name:'meituan'}];
    $scope.status = {
        isOpen:true,
        isDisabled:true
    };
    $scope.doClick = function (evt) {
        evt.stopPropagation();
        $scope.status.isOpen = !$scope.status.isOpen
    };
    $scope.currentUser = {id:1,name:'Jerry'};
    $scope.selectItem = function (item) {
        $scope.currentUser = item;
    }
}]);
