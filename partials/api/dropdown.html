dropdown

Description

下拉菜单组件，支持多列。可以通过Provider设置多列数目，当列表项的数目大于多列数目时以多列显示

Usage

<div class="btn-group" uix-dropdown is-open="status.isopen">
    <button type="button" uix-dropdown-toggle ng-disabled="disabled">
        Button dropdown
    </button>
    <ul class="dropdown-menu" role="menu">
        <li><a href="#">Something else here</a></li>
        <li><a href="#">Separated link</a></li>
    </ul>
</div>

Restrict

  • 'AE'

Provider

  • uixDropdownProvider：全局配置dropdown相关的设置
  • methods
    • setColsNum(num)：设置多列数目

Arguments

ParamTypeDefaultDetail
isOpen false是否开启列表
colsNum 3设置列数,属性的配置会覆盖全局provider的配置,全局的配置会覆盖默认的配置
onToggle 打开关闭的时候的回调函数,含有一个布尔值的参数,表示是否是打开的状态

<div ng-controller="dropdownDemoCtrl">
    <!-- Simple dropdown -->
    <span class="dropdown" uix-dropdown on-toggle="toggled(open)">
      <a href class="dropdown-toggle" uix-dropdown-toggle>
          Click me for a dropdown, yo!
      </a>
      <ul class="dropdown-menu">
          <li ng-repeat="choice in items">
              <a href>{{choice}}</a>
          </li>
      </ul>
    </span>
    <!-- Single button -->
    <div class="btn-group" uix-dropdown is-open="status.isopen">
        <button type="button" class="btn btn-primary dropdown-toggle" uix-dropdown-toggle ng-disabled="disabled">
            Button dropdown <span class="caret"></span>
        </button>
        <ul class="dropdown-menu" role="menu">
            <li><a href="#">Something else here</a></li>
            <li class="divider"></li>
            <li><a href="#">Separated link</a></li>
        </ul>
    </div>
    <!-- multi cols button -->
    <div class="btn-group" uix-dropdown>
        <button type="button" class="btn btn-primary dropdown-toggle" uix-dropdown-toggle>
            Multi cols dropdown <span class="caret"></span>
        </button>
        <ul class="dropdown-menu" role="menu">
            <li><a href="#">Action</a></li>
            <li><a href="#">Another action</a></li>
            <li><a href="#">Something else here</a></li>
            <li class="divider"></li>
            <li><a href="#">Separated link</a></li>
        </ul>
    </div>
    <!-- Split button -->
    <div class="btn-group" uix-dropdown>
        <button type="button" class="btn btn-danger">Action</button>
        <button type="button" class="btn btn-danger dropdown-toggle" uix-dropdown-toggle>
            <span class="caret"></span>
            <span class="sr-only">Split button!</span>
        </button>
        <ul class="dropdown-menu" role="menu">
            <li><a href="#">Action</a></li>
            <li><a href="#">Another action</a></li>
            <li><a href="#">Something else here</a></li>
            <li class="divider"></li>
            <li><a href="#">Separated link</a></li>
        </ul>
    </div>
    <hr />
    <p>
        <button type="button" class="btn btn-default btn-sm" ng-click="toggleDropdown($event)">Toggle button dropdown</button>
        <button type="button" class="btn btn-warning btn-sm" ng-click="disabled = !disabled">Enable/Disable</button>

</div>
angular.module('uixDemo').config(['uixDropdownProvider', function (uixDropdownProvider) {
    uixDropdownProvider.setColsNum(3);
}]);
angular.module('uixDemo').controller('dropdownDemoCtrl', ['$scope', '$log', function ($scope, $log) {
    $scope.items = [
        'The first choice!',
        'And another choice for you.',
        'but wait! A third!'
    ];

    $scope.status = {
        isopen: false
    };

    $scope.toggled = function (open) {
        $log.log('Dropdown is now: ', open);
    };

    $scope.toggleDropdown = function ($event) {
        $event.preventDefault();
        $event.stopPropagation();
        $scope.status.isopen = !$scope.status.isopen;
    };
}]);
