switch

Description

开关组件，用于开启某些状态的地方。

Usage

<fugu-switch ng-model="boolean"
    [ size="{{string}}" ]
    [ type="{{string}}" ]
    [ true-value="value" ]
    [ false-value="value" ]
    [ on-change="function" ]>
    </fugu-switch>

Restrict

  • 'E'

Arguments

ParamTypeDefaultDetail
ngModel 绑定数据，必须赋值
size(optional) "md"大小。可取值sm,md,lg
type(optional) "default"开关类型。可取值default,primary,success,info,error
trueValue(optional)   true当选中的时候ngModal的值
falseValue(optional)   false当未选中的时候ngModal的值
onChange(optional) 点击之后如果ngModel绑定对象改变,触发该方法

Example

  • index.html
  • script.js
<div ng-controller="switchDemoCtrl">
    <p>Normal</p>
    <fugu-switch ng-model="checked"></fugu-switch>
    <input type="checkbox" ng-model="checked"/>checked:{{checked}}
    <p>Disabled</p>
    <fugu-switch ng-model="open" ng-disabled="isDisabled"></fugu-switch>
    <p>Size</p>
    <fugu-switch ng-model="checked" size="sm"></fugu-switch>
    <fugu-switch ng-model="checked" type="success"></fugu-switch>
    <fugu-switch ng-model="checked" size="lg"></fugu-switch>
    <p>Type</p>
    <fugu-switch ng-model="open"></fugu-switch>
    <fugu-switch ng-model="open" type="primary"></fugu-switch>
    <fugu-switch ng-model="open" type="success"></fugu-switch>
    <fugu-switch ng-model="open" type="error"></fugu-switch>
    <fugu-switch ng-model="open" type="info"></fugu-switch>
    <p>Changed</p>
    <fugu-switch ng-model="open" on-change="statusChange()" type="success"></fugu-switch>
    <p>trueValue and falseValue</p>
    <fugu-switch ng-model="status" type="success" true-value="A" false-value="B"></fugu-switch>
    <pre>status:{{status}}</pre>
</div>
angular.module('fuguDemo').controller('switchDemoCtrl',['$scope', function ($scope) {
    $scope.open = false;
    $scope.checked = true;
    $scope.isDisabled = true;
    $scope.statusChange = function () {
        console.log('changed');
    };
    $scope.status = 'A';
    $scope.A = 'A';
    $scope.B = 'B';
}]);
